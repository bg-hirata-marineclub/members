// ========================================== // 💡 スタッフ権限確認 ＆ 既存スタッフデータの取得 // ========================================== function verifyStaff(uid) { try { var ss = SpreadsheetApp.openById(CONFIG.SPREADSHEET_ID); var memberSheet = ss.getSheetByName(CONFIG.SHEET_NAME_MEMBER); var mData = memberSheet.getDataRange().getValues(); var mHeaders = mData[0]; var uidCol = mHeaders.indexOf("LINE UID") !== -1 ? mHeaders.indexOf("LINE UID") : mHeaders.indexOf("LINE_UID"); var roleCol = mHeaders.indexOf("権限"); var certCol = mHeaders.indexOf("保有資格"); var lastNameCol = mHeaders.indexOf("姓"); var firstNameCol = mHeaders.indexOf("名"); var idCol = mHeaders.indexOf("会員ID"); var memberId = ""; var staffName = ""; var certs = ""; var isStaff = false; for (var i = 1; i < mData.length; i++) { if (mData[i][uidCol] === uid) { if (roleCol !== -1 && mData[i][roleCol] === "スタッフ") { isStaff = true; memberId = mData[i][idCol]; staffName = mData[i][lastNameCol] + " " + mData[i][firstNameCol]; certs = (certCol !== -1) ? mData[i][certCol] : ""; break; } else { return { status: "error", message: "スタッフ権限がありません。" }; } } } if (!isStaff) return { status: "error", message: "会員名簿に登録が見つかりません。" }; var staffSheet = ss.getSheetByName("M_スタッフ"); var staffDataObj = null; if (staffSheet) { var sData = staffSheet.getDataRange().getValues(); var sHeaders = sData[0]; var sIdCol = sHeaders.indexOf("会員ID"); for (var j = 1; j < sData.length; j++) { if (sData[j][sIdCol] === memberId) { staffDataObj = {}; for (var k = 0; k < sHeaders.length; k++) { staffDataObj[sHeaders[k]] = (sData[j][k] instanceof Date) ? Utilities.formatDate(sData[j][k], "JST", "yyyy-MM-dd") : sData[j][k]; } break; } } } return { status: "success", staffName: staffName, certs: certs, memberId: memberId, staffData: staffDataObj }; } catch (e) { return { status: "error", message: e.toString() }; } } // ========================================== // 💡 勤怠打刻記録（日報コメント対応版） // ========================================== function processTimecard(data) { var lock = LockService.getScriptLock(); if (!lock.tryLock(10000)) return { status: "error", message: "混雑しています" }; try { var ss = SpreadsheetApp.openById(CONFIG.SPREADSHEET_ID); var sheet = ss.getSheetByName("T_勤怠記録"); if (!sheet) { sheet = ss.insertSheet("T_勤怠記録"); sheet.appendRow(["タイムスタンプ", "会員ID", "氏名", "打刻種類", "船舶手当", "打刻場所（GPS）", "日報コメント"]); sheet.getRange("A1:G1").setFontWeight("bold").setBackground("#f3f3f3"); } var now = Utilities.formatDate(new Date(), "JST", "yyyy/MM/dd HH:mm"); var mapUrl = (data.lat && data.lng) ? "https://www.google.com/maps?q=" + data.lat + "," + data.lng : "位置情報なし"; // 💡修正: data.comment をG列（7番目）として保存 sheet.appendRow([now, data.memberId, data.staffName, data.type, data.shipAllowance, mapUrl, data.comment || ""]); return { status: "success" }; } catch (e) { return { status: "error", message: e.toString() }; } finally { lock.releaseLock(); } } // ========================================== // 💡 スタッフ情報（口座・免許）の登録・更新 // ========================================== function registerStaff(data) { var lock = LockService.getScriptLock(); if (!lock.tryLock(10000)) return { status: "error", message: "混雑しています" }; try { var ss = SpreadsheetApp.openById(CONFIG.SPREADSHEET_ID); var sheet = ss.getSheetByName("M_スタッフ"); if (!sheet) { sheet = ss.insertSheet("M_スタッフ"); sheet.appendRow(["登録日時", "会員ID", "氏名", "銀行名", "支店名", "口座種別", "口座番号", "口座名義(カナ)", "免許画像URL", "有効期限"]); sheet.getRange("A1:J1").setFontWeight("bold").setBackground("#f3f3f3"); } var sData = sheet.getDataRange().getValues(); var headers = sData[0]; var idCol = headers.indexOf("会員ID"); var targetRow = -1; var existingImageUrl = ""; for (var i = 1; i < sData.length; i++) { if (sData[i][idCol] === data.memberId) { targetRow = i + 1; existingImageUrl = sData[i][headers.indexOf("免許画像URL")]; break; } } var now = Utilities.formatDate(new Date(), "JST", "yyyy/MM/dd HH:mm"); var fileUrl = existingImageUrl; if (data.licenseImageBase64 && data.licenseImageBase64.indexOf("data:image") === 0) { var folder = DriveApp.getFolderById("1jhWYLbcMKHTuViDmaZcaD3Dhd86IJ1ez"); var contentType = data.licenseImageBase64.match(/data:([a-zA-Z0-9]+\/[a-zA-Z0-9-.+]+).*,/)[1]; var base64Data = data.licenseImageBase64.split(',')[1]; var fileName = data.memberId + "_" + data.staffName + "_船舶免許"; var blob = Utilities.newBlob(Utilities.base64Decode(base64Data), contentType, fileName); fileUrl = folder.createFile(blob).getUrl(); } if (targetRow !== -1) { sheet.getRange(targetRow, headers.indexOf("登録日時") + 1).setValue(now); sheet.getRange(targetRow, headers.indexOf("銀行名") + 1).setValue(data.bankName); sheet.getRange(targetRow, headers.indexOf("支店名") + 1).setValue(data.branchName); sheet.getRange(targetRow, headers.indexOf("口座種別") + 1).setValue(data.accountType); sheet.getRange(targetRow, headers.indexOf("口座番号") + 1).setValue(data.accountNumber); sheet.getRange(targetRow, headers.indexOf("口座名義(カナ)") + 1).setValue(data.accountName); sheet.getRange(targetRow, headers.indexOf("免許画像URL") + 1).setValue(fileUrl); sheet.getRange(targetRow, headers.indexOf("有効期限") + 1).setValue(data.licenseExpiry); } else { var newRow = []; for (var k = 0; k < headers.length; k++) { var h = headers[k]; if (h === "登録日時") newRow.push(now); else if (h === "会員ID") newRow.push(data.memberId); else if (h === "氏名") newRow.push(data.staffName); else if (h === "銀行名") newRow.push(data.bankName); else if (h === "支店名") newRow.push(data.branchName); else if (h === "口座種別") newRow.push(data.accountType); else if (h === "口座番号") newRow.push(data.accountNumber); else if (h === "口座名義(カナ)") newRow.push(data.accountName); else if (h === "免許画像URL") newRow.push(fileUrl); else if (h === "有効期限") newRow.push(data.licenseExpiry); else newRow.push(""); } sheet.appendRow(newRow); } return { status: "success" }; } catch (e) { return { status: "error", message: e.toString() }; } finally { lock.releaseLock(); } }
